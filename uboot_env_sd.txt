ethaddr=00:01:02:03:04:05
bootdelay=5
autostart=n
autoload=n
binaddr=0x30000000
envaddr=0x30100000
ubootfile=qspiboot.bin
ubootenv=qspiboot.env
bootcmd= echo "Writing Flash Rom" &&
   fatload mmc 0:1 ${binaddr} ${ubootfile} && 
   fatloat mmc 0:1 ${envaddr} ${ubootenv} &&
   sf probe 0 0 0 && 
   sf write ${binaddr} 0x0 0x100000 &&
   sf write ${envaddr} 0x100000 0x20000 &&
   echo "QSPI programming completed.. "
